# Run 'make', then 'touch one.h' and 'make' again.
#defines the CC variable
CC=cc

#target all dependent on one file
all: one

#one target dependent on one.o main.o two.o three.o
#with a recipe to compile the object files 
one: one.o main.o two.o three.o
	${CC} one.o main.o two.o three.o -o one

#target one.o dependent on ext.h one.h and one.c
#with a recipe to compile the c file one.c
one.o: ext.h one.h one.c
	${CC} -c one.c

#target main.o dependent on ext.h one.h main.c
#with recipe to compile main.c
main.o: ext.h one.h main.c
	${CC} -c main.c

#target two.o dependent on one.h and two.c
#recipe to compile two.c
two.o: one.h two.c
	${CC} -c two.c

#target three.o dependent on one.h and three.c
#recipe to compile three.c
three.o: one.h three.c
	${CC} -c three.c

#sudo target clean
#recipe to remove .o files
clean:
	rm -f one *.o
