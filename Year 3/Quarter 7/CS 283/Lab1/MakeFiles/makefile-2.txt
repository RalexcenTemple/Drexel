#defines prog variable
PROG=	test

#defines OBJS variable to be a collection of object files
OBJS=	one.o main.o two.o three.o

#defines the CFLAGS varaible
CFLAGS=	-Wall -g

#all target with dependency on the value of the PROG variable
all: ${PROG}

#target value of varaible Prog wit hdependency on value of objs variable
#with recipe to echo the dependenies and then
#compile the objs files
#LDFLAGS serves no purpose here 
${PROG}: ${OBJS}
	@echo $@ depends on $?
	${CC} ${LDFLAGS} ${OBJS} -o ${PROG}

#.c.o target
#recipe to compile using the cflags and the first prerequiste of the target
#file
.c.o:
	$(CC) -c $(CFLAGS) -o $@ $<


#sudo target clean w/out prerequisites
#recipe to remove the object files	
clean:
	rm -f ${PROG} ${OBJS}
